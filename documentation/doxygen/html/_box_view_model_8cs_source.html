161 public BoxViewModel()
162 {
163 _boxRepository = new BoxRepository();
164 _materialRepository = new MaterialRepository();
165
166 Boxes = new ObservableCollection<Box>(_boxRepository.GetAll());
167 MaterialsAvailable = new ObservableCollection<Material>(_materialRepository.GetAll());
168 // In the materials available list, remove materials that are already in boxes
169 if (Boxes != null && MaterialsAvailable != null)
170 {
171 var guidsToRemove = Boxes
172 .Where(box => box.MaterialGuids != null)
173 .SelectMany(box => box.MaterialGuids)
174 .ToHashSet();
175
176 var materialsToRemove = MaterialsAvailable
177 .Where(material => guidsToRemove.Contains(material.GUID))
178 .ToList();
179
180 foreach (var material in materialsToRemove)
181 MaterialsAvailable.Remove(material);
182 }
183
184 _newBoxName = string.Empty;
185 _newBoxDescription = string.Empty;
186
187 AddBoxCommand = new RelayCommand(AddBox, CanAddBox);
188 RemoveBoxCommand = new RelayCommand(RemoveBox, CanRemoveBox);
189 UpdateBoxCommand = new RelayCommand(SaveBox, CanSaveBox);
190 AddMaterialInBoxCommand = new RelayCommand(AddMaterialInBox, CanAddInFormMaterialInBox);
191 RemoveMaterialInBoxCommand = new RelayCommand(RemoveMaterialInBox, CanRemoveInFormMaterialInBox);

193 SetButtonVisibility();
